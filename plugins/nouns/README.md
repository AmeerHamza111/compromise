```js
const nlp = require('compromise')
nlp.extend(require('compromise-nouns'))

let doc = nlp('The Children are right to laugh at you, Ralph')
doc.nouns().debug()
// 'children'
// 'laugh'
// 'you'
// 'Ralf'
```
